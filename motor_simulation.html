# 
G_H = tf(Kp, 1)
G_I = tf(Ki, [1, 0])
G_J = tf([Kd, 0], [Tf, 1])
G_K = tf(1, [L, R])
G_L = tf(Kt, 1)
G_M = tf(1, [J, Kf])
G_N = tf(Ke, 1)

# omegar --> omega
G_omegar2omega = \
    (
        (G_H + G_I + G_J) * G_K * G_L * G_M / (1 + G_K * G_L * G_M * G_N)
    ) / (
        1 + G_K * G_L * G_M * (G_H + G_I + G_J) / (1 + G_K * G_L * G_M * G_N)
    )

# tauL --> omega
G_tauL2omega = \
    (
        G_M / (1 + G_K * G_L * G_M * G_N)
    ) / (
        1 + G_K * G_L * G_M * (G_H + G_I + G_J)
        / (1 + G_K * G_L * G_M * G_N)
    )
